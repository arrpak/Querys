select company,name
from fn3newsstand
where code in
(
select newsstand
from fn3newsstandon
where
substr(newsstand,1,3)='028' and
(dateoff is null or dateoff >sysdate)
minus
(
select newsstand
from fn3operation
where
substr(newsstand,1,3)='028' and
operationdate=to_date('20030520','yyyymmdd')));
