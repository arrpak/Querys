Select servicio.newsstand,service,decision
from
((select newsstand,service
from fn3operation
where header='MARCA' and product='DIARIO'
and   operationdate=to_date('20030520','yyyymmdd')
and   service=0
and   substr(newsstand,1,3)='028'
union
(select newsstand,service
from fn3operationnotcontrolled
where header='MARCA' and product='DIARIO'
and   service=0
and   operationdate=to_date('20030520','yyyymmdd')
and   substr(newsstand,1,3)='028'))) servicio,
(select newsstand,decision,orderf
from fn3printingdetail
where
header='MARCA' and product='DIARIO'
and   dateprint=to_date('20030520','yyyymmdd')
and decision>0
and node='CAM'
and   substr(newsstand,1,3)='028') tirada,
(select orderf
 from fn3printing
where
header='MARCA' and product='DIARIO' 
and dateprint=to_date('20030520','yyyymmdd')
and used='Y'
and node='CAM') orden
where
servicio.newsstand=tirada.newsstand and
tirada.orderf=orden.orderf
;
