
--BUSCA LA TIRADA DE MAYOR ORDEN Y DICE EL ESTADO DEL CAMPO USADA
SELECT T1.DATEPRINT,T1.USED,T2.ORDERF
FROM EMUPRINTING T1,
(
SELECT DATEPRINT,Max(ORDERF) AS ORDERF
FROM EMUPRINTING
WHERE
NODE='CAMCAP' AND PRODUCT='DIARIO'  AND DATEPRINT>To_Date('2011/01/01','YYYY/MM/DD')
--AND USED='N'
GROUP BY DATEPRINT
) T2
WHERE
NODE='CAMCAP' AND PRODUCT='DIARIO'  AND T1.DATEPRINT>To_Date('2011/02/21','YYYY/MM/DD')
--AND T1.NODE=T2.NODE
--AND T1.PRODUCT=T2.PRODUCT
AND T1.DATEPRINT=T2.DATEPRINT
AND T1.ORDERF=T2.ORDERF
--AND USED='N'
GROUP BY T1.DATEPRINT,T1.USED,T2.ORDERF





--SACA LA FORSESSION PARA LA TIRADA USADA
SELECT DISTINCT T1.DATEPRINT,T1.DECSESSION
FROM EMUPRINTINGNODEDECSESSION T1,EMUPRINTING T2
WHERE
T2.USED='Y'
AND T1.DATEPRINT>To_Date('2011/01/01','YYYY/MM/DD')
AND T1.DATEPRINT=T2.DATEPRINT
AND T1.ORDERF=T2.ORDERF
