SELECT *
from
(SELECT NEWSSTAND
FROM FN3NODE
WHERE HEADER='MARCA' AND PRODUCT='DIARIO'
AND FATHERNODE='ANDJAESIE') T1,
( SELECT NEWSSTAND,(Sum(RETURN)/SUM(sERVICE)) AS PorDev
FROM fn3operation
WHERE HEADER='MARCA' AND PRODUCT='DIARIO'
AND operationdate>SYSDATE-20
GROUP BY newsstand
ORDER BY PorDev ) t2
WHERE
t1.newsstand=t2.newsstand
