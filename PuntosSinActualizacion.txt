INSERT INTO FN3PRINTINGDETAIL
SELECT HEADER,PRODUCT,NODE,DATEPRINT,2,NEWSSTAND,FORSESSION,DECISION,INCREASEDEC
FROM FN3PRINTINGDETAIL
where header='MARCA' and
Product='DIARIO' AND
DATEPRINT=TO_DATE('20031029','YYYYMMDD') AND
NODE ='CAVCMU' AND
ORDERF=1 AND
NEWSSTAND IN
(
select newsstand
from fn3printingdetail
where header='MARCA' and
Product='DIARIO' AND
DATEPRINT=TO_DATE('20031029','YYYYMMDD') AND
NODE ='CAVCMU' AND
ORDERF=1 
MINUS
(select newsstand
from fn3printingdetail
where header='MARCA' and
Product='DIARIO' AND
DATEPRINT=TO_DATE('20031029','YYYYMMDD') AND
NODE ='CAVCMU' AND
ORDERF=2));
