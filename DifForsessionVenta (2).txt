SELECT SUBSTR(OUTPUT,11,12),VALUEDATE,FORECAST,OPE.VENTA
FROM
(SELECT DISTINCT(FORSESSION),DATEPRINT
FROM FN3PRINTINGNODEFORSESSION
WHERE DATEPRINT=TO_DATE('20030502','YYYYMMDD')
AND HEADER='MARCA' AND
PRODUCT='DIARIO'
AND NODE='CAM' AND ORDERF=
(SELECT ORDERF
FROM FN3PRINTING
WHERE NODE='CAM' AND
HEADER='MARCA' AND
PRODUCT='DIARIO' AND
USED='Y' AND
DATEPRINT=TO_DATE('20030502','YYYYMMDD'))) F,
FN3FORECASTSERIES SER,
(SELECT NEWSSTAND,OPERATIONDATE,SERVICE+INCIDENCE-RETURN AS VENTA
FROM FN3OPERATION
WHERE
HEADER='MARCA' AND
PRODUCT='DIARIO' AND
SUBSTR(NEWSSTAND,1,3)='028' AND
OPERATIONDATE=TO_DATE('20030502','YYYYMMDD')) OPE
WHERE
SER.FORSESSION=f.FORSESSION AND
SER.DECISION>0 AND
F.DATEPRINT=SER.VALUEDATE AND
OPE.NEWSSTAND=SUBSTR(OUTPUT,11,12);
SPOOL OFF;
